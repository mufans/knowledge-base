    3. ж ёеҝғз»„д»¶¶

    3.1 LangGraph е·ҘдҪңжөҒ¶

    дҪҝз”Ё StateGraph е®ҡд№үжңүеҗ‘еӣҫпјҢиҠӮзӮ№жҳҜ AgentпјҢиҫ№жҳҜж•°жҚ®жөҒе’ҢжҺ§еҲ¶жөҒгҖӮ

    workflow = StateGraph(AgentState)

# иҠӮзӮ№
workflow.add_node("requirement_analyst", ...)
workflow.add_node("architect", ...)
workflow.add_node("coder", ...)
workflow.add_node("reviewer", ...)

# иҫ№
workflow.set_entry_point("requirement_analyst")
workflow.add_edge("requirement_analyst", "architect")
workflow.add_edge("architect", "coder")
workflow.add_edge("coder", "reviewer")

# жқЎд»¶иҫ№пјҲеҸҚйҰҲеҫӘзҺҜпјү
workflow.add_conditional_edges("reviewer", should_continue, {
    "coder": "coder",   # е®ЎжҹҘдёҚйҖҡиҝҮ вҶ’ йҮҚж–°з”ҹжҲҗ
    END: END,            # е®ЎжҹҘйҖҡиҝҮ вҶ’ з»“жқҹ
})

    3.2 Agent зҠ¶жҖҒпјҲAgentStateпјү¶

    жүҖжңү Agent е…ұдә«дёҖдёӘ TypedDict зҠ¶жҖҒпјҢйҖҡиҝҮ LangGraph зҡ„ reducer жңәеҲ¶иҮӘеҠЁеҗҲе№¶жӣҙж–°гҖӮ

    class AgentState(TypedDict):
    user_input: str                    # еҺҹе§ӢйңҖжұӮ
    platform: str                      # зӣ®ж Үе№іеҸ°
    requirement: dict                  # йңҖжұӮеҲҶжһҗз»“жһң
    architecture: dict                 # жһ¶жһ„и®ҫи®Ў
    generated_files: list[dict]        # з”ҹжҲҗзҡ„д»Јз Ғж–Үд»¶
    review_result: dict                # е®ЎжҹҘз»“жһң
    test_files: list[dict]             # жөӢиҜ•ж–Үд»¶
    iteration_count: int               # иҝӯд»Јж¬Ўж•°пјҲжҺ§еҲ¶еҫӘзҺҜпјү
    messages: list[dict]               # ж¶ҲжҒҜеҺҶеҸІ
    errors: list[str]                  # й”ҷиҜҜи®°еҪ•
    status: str                        # зҠ¶жҖҒ

    3.3 Agent иҒҢиҙЈ¶

    Agent иҫ“е…Ҙ иҫ“еҮә е…ій”®иғҪеҠӣ
    йңҖжұӮеҲҶжһҗ иҮӘз„¶иҜӯиЁҖйңҖжұӮ з»“жһ„еҢ–йңҖжұӮ JSON йңҖжұӮжӢҶи§ЈгҖҒе®Ңж•ҙжҖ§жЈҖжҹҘгҖҒйҡҗејҸйңҖжұӮжҺЁж–ӯ
    жһ¶жһ„и®ҫи®Ў з»“жһ„еҢ–йңҖжұӮ жһ¶жһ„ж–№жЎҲ JSON жЁЎеқ—еҲ’еҲҶгҖҒеҲҶеұӮи®ҫи®ЎгҖҒдҫқиө–з®ЎзҗҶ
    д»Јз Ғз”ҹжҲҗ жһ¶жһ„ж–№жЎҲ д»Јз Ғж–Үд»¶еҲ—иЎЁ еӨҡж–Үд»¶з”ҹжҲҗгҖҒе№іеҸ°йҖӮй…ҚгҖҒжңҖдҪіе®һи·ө
    д»Јз Ғе®ЎжҹҘ д»Јз Ғж–Үд»¶ е®ЎжҹҘз»“жһң JSON йқҷжҖҒеҲҶжһҗгҖҒиҙЁйҮҸиҜ„еҲҶгҖҒдҝ®ж”№е»әи®®
